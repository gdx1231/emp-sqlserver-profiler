# emp-sqlserver-profiler
Java sqlserver profiler
# Usage (linux/mac):
java -cp target/emp-sqlserver-profiler-1.0.0.jar:target/lib/* com.gdxsoft.sqlProfiler.ProfilerControl -h 192.168.1.100 -u sa -p yourpassword -d master -P 1433
# Usage (windows):
java -cp target/emp-sqlserver-profiler-1.0.0.jar;target/lib/* com.gdxsoft.sqlProfiler.ProfilerControl -h 192.168.1.100 -u sa -p yourpassword -d master -P 1433
